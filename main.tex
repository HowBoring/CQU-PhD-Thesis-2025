    \chapter[\hspace{0pt}#1]{\heiti\zihao{3}\hspace{0pt}#2}%
  \fi
}

\providecommand{\mysection}[2][]{%
  \if\relax\detokenize{#1}\relax

  \else
    \section[\hspace{-2pt}#1]{\heiti\zihao{-3}\hspace{-8pt}#2}%

}

}
